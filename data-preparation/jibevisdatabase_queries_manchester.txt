SELECT 
    AVG(t."t.distance_walk") "t.distance_walk",
    stddev(t."t.distance_walk") "t.distance_walksd",
    AVG(t."t.distance_bike") "t.distance_bike", 
    stddev(t."t.distance_bike") "t.distance_bikesd", 
    AVG(t."t.distance_auto")  "t.distance_auto",
    stddev(t."t.distance_auto")  "t.distance_autosd"
FROM (
SELECT 
    personid,
        SUM(b."t.distance_walk") "t.distance_walk",
        SUM(b."t.distance_bike") "t.distance_bike", 
        SUM(b."t.distance_auto")  "t.distance_auto"
    FROM synpop_manchester_2021 a
    LEFT JOIN trips_manchester b 
     ON a.personid = b."p.id"
     WHERE a.age >= 18 AND scenario = 'reference'
     GROUP BY personid
 ) t;
